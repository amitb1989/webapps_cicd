#!/bin/bash
set -eux

# Update & install Python and Git
dnf -y update
dnf -y install python3-pip git

# Create app user
id webapp >/dev/null 2>&1 || useradd -m -s /bin/bash webapp

# App directory structure
mkdir -p /home/webapp/{templates,static}
chown -R webapp:webapp /home/webapp

# --- Flask app: app.py ---
cat > /home/webapp/app.py << 'PY'
from flask import Flask, render_template
app = Flask(__name__)

@app.route("/")
def index():
    # Renders templates/index.html
    return render_template("index.html"), 200

@app.route("/healthz")
def healthz():
    return "ok", 200

if __name__ == "__main__":
    # Running on port 80 as root due to systemd config; keep 0.0.0.0 for ALB
    app.run(host="0.0.0.0", port=80)
PY

# Install Flask (pin a stable release)
pip3 install "flask>=2,<4"

# Systemd service (run as root to bind port 80; for least privilege, see note below)
cat > /etc/systemd/system/webapp.service << 'UNIT'
[Unit]
Description=HA Flask WebApp
After=network.target

[Service]
User=root
WorkingDirectory=/home/webapp
ExecStart=/usr/bin/python3 /home/webapp/app.py
Restart=always
RestartSec=2
Environment=PYTHONUNBUFFERED=1

[Install]
WantedBy=multi-user.target
UNIT

# Enable & start
systemctl daemon-reload
systemctl enable webapp
systemctl start webapp

# --- Install CodeDeploy agent (robust region detection, no jq needed) ---
dnf -y install ruby wget

# Try IMDSv2 for region; fall back to IMDSv1 AZ->region if needed
IMDS=http://169.254.169.254
TOKEN=$(curl -s -X PUT "${IMDS}/latest/api/token" -H "X-aws-ec2-metadata-token-ttl-seconds: 21600" || true)

if [ -n "$TOKEN" ]; then
  # IMDSv2
  az=$(curl -s -H "X-aws-ec2-metadata-token: $TOKEN" "${IMDS}/latest/meta-data/placement/availability-zone" || true)
else
  # IMDSv1 fallback
  az=$(curl -s "${IMDS}/latest/meta-data/placement/availability-zone" || true)
fi

# Convert AZ to region (e.g., ap-south-1a -> ap-south-1)
region="${az::-1}"

# Final sanity fallback (hardcode your region if detection failed)
if [ -z "$region" ]; then
  region="ap-south-1"
fi

echo "Resolved region: $region"

cd /home/ec2-user
installer_url="https://aws-codedeploy-${region}.s3.${region}.amazonaws.com/latest/install"
echo "Fetching CodeDeploy installer from: $installer_url"
wget -O install "$installer_url"
chmod +x ./install
./install auto

systemctl enable codedeploy-agent
systemctl start codedeploy-agent
systemctl status codedeploy-agent --no-pager || true
